randomSearch
open /index.php
store 1 num
while ${num}<=2
store javascript{"term"+ Math.random().toString(36).substring(7)} searchTerm
click id=search_query_top
type id=search_query_top ${searchTerm}
clickAndWait name=submit_search
pause 2000
assertText css=.alert.alert-warning No results were found for your search "${searchTerm}"
storeEval ${num}+1 num
echo ${num}
endWhile
